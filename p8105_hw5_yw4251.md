sim_mean_test <- function(n=30, mu, sigma = 5) {
  sim_data = tibble(
    x=rnorm(n=n, mu, sd = sigma)
  )
  test_result = broom::tidy(t.test(sim_data$x, mu=mu))|>
    select(mu_hat=estimate,p.value)
  return(test_result)
}

sim_mean_test(mu=0)
